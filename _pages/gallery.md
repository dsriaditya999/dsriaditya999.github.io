<script>
  // Function to add hover pause effect to swiper containers
  function addHoverPause(swiperId) {
    const swiperContainer = document.getElementById(swiperId);
    const swiperInstance = swiperContainer.swiper;

    swiperContainer.addEventListener('mouseenter', function () {
      swiperInstance.autoplay.stop();
    });

    swiperContainer.addEventListener('mouseleave', function () {
      swiperInstance.autoplay.start();
    });
  }

  document.addEventListener('DOMContentLoaded', function() {
    addHoverPause('highlights-swiper');
    addHoverPause('individuals-swiper');
    addHoverPause('projects-swiper');
    addHoverPause('personal-swiper');
  });
</script>
